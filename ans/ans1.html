//Reverse an array in its place

            #include <stdio.h>
            int main(){
                //inputing array
                int n;
                printf("enter size of array:");
                scanf("%d",&n);
                int a[n],temp;
                printf("enter array elements:");
                for(int i=0;i<n;i++)
                {
                    scanf("%d",&a[i]);
                }

                //reversing array
                for(int i=0;i<n/2;i++)
                {
                    temp=a[i];
                    a[i]=a[n-i-1];
                    a[n-i-1]=temp;
                }

                //printing reversed array
                for(int i=0;i<n;i++)
                {
                    printf("%d ",a[i]);
                }
                return 0;
            }
