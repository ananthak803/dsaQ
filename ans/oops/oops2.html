/*Friend class - A friend class can access private and protected members of 
            other classes in which it is declared as a friend.*/

            #include <iostream>
                using namespace std;
                
                class A
                {
                    int a = 1;
                    int b = 2;
                
                public:
                    friend class B;
                };
                
                class B
                {
                    A obj;
                public:
                
                    /*accessing variables of a without passing parameter in function , and object obj of 
                    class A is created.*/
                    void sum()
                    {
                        cout << obj.a<<endl;
                    }
                    
                    //this function will be called when object is passed as parameter.
                    void sum(A obj)
                    {
                        cout << obj.b<<endl;
                    }
                };
                
                int main()
                {
                    B x;
                    x.sum();
                    A y;
                    x.sum(y);
                    return 0;
                }
