/*The elements of integer array is the price of a given stock on the ith day.
            maximize your profit by choosing a single day to buy one stock and choosing a 
            different day in the future to sell that stock.print profit*/
            
            /*leetcode link : https://leetcode.com/problems/best-time-to-buy-and-sell-stock/description/  */
            
            #include <stdio.h>
            #include <limits.h>  //to use INT_MAX
            void main()
            {
                //declarations
                int n,i,max=0,min=INT_MAX;
            
                //array size input
                printf("enter size of array:");
                scanf("%d",&n);
            
                //array input
                int a[n];
                printf("enter array elements:");
                for(i=0;i<n;i++)
                    scanf("%d",&a[i]);    
            
                //finding max profit
                for(i=0;i<n;i++)
                {
                    min=min>a[i]?a[i]:min;
                    max=max<a[i]-min?a[i]-min:max;
                }
            
                //printing result
                if(max==0)
                    printf("0 profit");
                else
                    printf("profit:%d",max);
            
            }
