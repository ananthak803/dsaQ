#include <stdio.h>
            #define max 5
                
                int enqueue(int array[],int* rear,int* front,int item)
                {
                    if(*rear==max-1)
                    {
                        printf("queue is full\n");
                        return 1;
                    }
                    else
                    {
                        if(*rear==-1)
                        {
                            *rear=0;
                            *front=0;
                        }
                        else
                        {
                            (*rear)++;
                        }
                        array[(*rear)]=item;
                        
                    }
                    
                }
                
                int dequeue(int array[],int* rear,int* front)
                {
                    if(*rear==-1)
                    {
                        printf("queue is empty\n");
                        return 1;
                    }
                    else
                    {
                        printf("element removed is %d \n",array[*front]);
                        (*front)++;
                        if(*rear<*front)       //resetting queue if empty
                        {
                            *rear=-1;
                            *front=-1;
                        }
                    }
                }
                
                int display(int array[],int *rear,int *front)
                {
                    if(*rear==-1)
                    {
                        printf("queue is empty\n");
                        return 1;
                    }
                    for(int i=*front;i<=*rear;i++)
                    {
                        printf("%d ",array[i]);
                    }
                }
                
                int main()
                {
                int a[max],item,key=1,rear=-1,front=-1;
                while(key)
                {
                    printf("\nenqueue(1)|dequeue(2)|display(3)|exit(0)");
                    scanf("%d",&key);
                    switch(key)
                    {
                        case 1:
                                printf("enter element to insert:");
                                scanf("%d",&item);
                                enqueue(a,&rear,&front,item);
                                break;
                        case 2:
                                dequeue(a,&rear,&front);
                                break;
                        case 3:
                                display(a,&rear,&front);
                                break;
                        case 0:
                                return 0;
                            
                    }
                }
                }
