/*Program to reverse a queue*/
        
            #include <stdio.h>
            #define max 10
                
                //reverse fucntion
                void reverse(int a[], int *rear, int *front)
                {
                    int c = (*rear - *front + 1) / 2;
                    int r = *rear, f = *front, temp;
                    while (c != 0)
                    {
                        temp = a[f];
                        a[f] = a[r];
                        a[r] = temp;
                        f++;
                        r--;
                        c--;
                    }
                }
                
                void enQueue(int a[], int *rear, int *front, int k)
                {
                    if (*rear == max - 1)
                    {
                        printf("overflow\n");
                        return;
                    }
                    else
                    {
                        if (*rear == -1)
                        {
                            *front = *rear = 0;
                            a[*rear] = k;
                            return;
                        }
                        a[++(*rear)] = k;
                    }
                }
                
                void deQueue(int *rear, int *front)
                {
                    if (*rear == -1)
                    {
                        printf("underflow\n");
                        return;
                    }
                    (*front)++;
                    if (*front > *rear)
                        *front = *rear = -1;
                }
                
                void display(int a[], int *rear, int *front)
                {
                    if (*rear == -1)
                        return;
                    for (int i = *front; i <= *rear; i++)
                        printf("%d ", a[i]);
                    printf("\n");
                }
                
                
                int main()
                {
                    int rear = -1, front = -1, a[max], k, choice = 1;
                    while (choice)
                    {
                        printf("enter choice: ");
                        scanf("%d", &choice);
                        switch (choice)
                        {
                        case 0:
                            return 0;
                        case 1:
                            printf("enter data:");
                            scanf("%d", &k);
                            enQueue(a, &rear, &front, k);
                            break;
                        case 2:
                            deQueue(&rear, &front);
                            break;
                        case 3:
                            display(a, &rear, &front);
                            break;
                        case 4:
                            reverse(a, &rear, &front);
                            break;
                        }
                    }
                }
