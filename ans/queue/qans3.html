/*Program to implement Deque i.e. double ended queue.*/

            #include <stdio.h>
            #define max 10
                
                void insertRear(int a[], int *rear, int *front, int item)
                {
                    if (*front == 0 && *rear == max - 1 || *front == *rear + 1)
                    {
                        printf("overflow\n");
                        return;
                    }
                    else if (*rear == -1)
                        *rear = *front = 0;
                    else if (*rear == max - 1)
                        *rear = 0;
                    else
                        (*rear)++;
                    a[*rear] = item;
                }
                
                void deleteRear(int *rear, int *front)
                {
                    if (*rear == -1)
                    {
                        printf("underflow\n");
                        return;
                    }
                    else if (*rear == 0)
                        *rear = max - 1;
                    else if (*rear == *front)
                        *rear = *front = -1;
                    else
                        (*rear)--;
                }
                
                void insertFront(int a[], int *rear, int *front, int item)
                {
                    if (*front == 0 && *rear == max - 1 || *front == *rear + 1)
                    {
                        printf("overflow\n");
                        return;
                    }
                    else if (*front == -1)
                        *front = *rear = 0;
                    else if (*front == 0)
                        *front = max - 1;
                    else
                        (*front)--;
                    a[*front] = item;
                }
                
                void deleteFront(int *rear, int *front)
                {
                    if (*front == -1)
                    {
                        printf("underflow\n");
                        return;
                    }
                    else if (*front == max - 1)
                        *front = 0;
                    else
                        (*front)++;
                }
                
                void isFull(int *rear,int *front)
                {
                    if(*front==0 && *rear==max-1 || *front==*rear+1)
                    {
                        printf("overflow\n");
                        return;
                    }
                }
                
                void isEmpty(int *rear)
                {
                    if(*rear==-1)
                    {
                        printf("underflow\n");
                        return;
                    }
                }
                
                void printRear(int a[],int *rear)
                {
                    if(*rear==-1)
                        return;
                    printf("%d\n",a[*rear]);
                }
                
                void printFront(int a[],int *front)
                {
                    if(*front==-1)
                        return;
                    printf("%d\n",a[*front]);
                }
                
                int main()
                {
                    int rear = -1, front = -1;
                    int a[max], choice = 1, k;
                
                    while (choice)
                    {
                        printf("enter choice:");
                        scanf("%d", &choice);
                        switch (choice)
                        {
                        case 0:
                            return 0;
                        case 1:
                            printf("enter data:");
                            scanf("%d", &k);
                            insertRear(a, &rear, &front, k);
                            break;
                        case 2:
                            printf("enter data:");
                            scanf("%d", &k);
                            insertFront(a, &rear, &front, k);
                            break;
                        case 3:
                            deleteRear(&rear, &front);
                            break;
                        case 4:
                            deleteFront(&rear, &front);
                            break;
                        case 5:
                            isFull(&rear, &front);
                            break;
                        case 6:
                            isEmpty(&rear);
                            break;
                        case 7:
                            printRear(a,&rear);
                            break;
                        case 8:
                            printFront(a,&front);
                            break;
                        }
                    }
                }
