//largest element in an array

            #include <stdio.h>
                int main(){
                    int n;
                    printf("enter n:");
                    scanf("%d",&n);
                    int a[n],max;
                    printf("enter array elements: ");
                    for(int i=0; i<n; i++)
                    {
                        scanf("%d",&a[i]);
                    }
                    max=a[0];
                    for(int i=0; i<n; i++)
                    {
                        if(max<a[i])
                        {
                            max=a[i];
                        }
                    }
                    printf("largest element: %d",max);
                    return 0;
                }
