//In an user inputed array "every element appears twice except for one". Find that single one.

            /*Leetcode link : https://leetcode.com/problems/single-number/description/ */
            
            #include <stdio.h>
            void main()
            {
                //declarations
                int i,key=0,n;
            
                //array size input
                printf("enter array size:");
                scanf("%d",&n);
            
                //array input
                int a[n];
                printf("enter array elements:");
                for(i=0;i<n;i++)
                    scanf("%d",&a[i]);
                
                //finding single element
                for(i=0;i<n;i++)
                    key=key^a[i];     //xor cancels pairs.  x^x=0 and x^0=x 
                
                //printing result
                printf("single element is %d",key);
            }
