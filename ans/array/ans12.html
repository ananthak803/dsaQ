/*Given a matrix of size n X n, where every row and every column is sorted in
        increasing order. Write an algorithm and a program to find whether the given key
        element is present in the matrix or not. (Linear time complexity)*/


        /*********** method 1 - without using UDF ***********/

        #include <stdio.h>
        int main()
        {
            //declarations
            int n,i,j,key;

            //array size input
            printf("enter size of array:");
            scanf("%d",&n);
            
            //2d array input
            int a[n][n];
            printf("enter array elements:");
            for(i=0; i<n; i++)
            {
                for(j=0; j<n; j++)
                {
                    scanf("%d",&a[i][j]);
                }
            }

            //key input
            printf("enter element to search:");
            scanf("%d",&key);

            // resetting value of i and j
            i=0,j=n-1;
            while(j>=0 && i<n)
            {
                if(a[i][j]==key)
                {
                    printf("present");
                    return 0;
                }
                else if(a[i][j]<key)
                {
                    i++;
                }
                else
                    j--;
            }
            printf("not present");
        }


        /*********** method 2 - using UDF ***********/

        #include <stdio.h>

        //function to search key in matrix of size n X n
        int search(int n,int array[n][n],int key)
        {
            int i=0,j=n-1;
            while(j>=0 && i<n)
            {
                if(array[i][j]==key)
                {
                    printf("present");
                    return 0;
                }
                else if(array[i][j]<key)
                {
                    i++;
                }
                else
                    j--;
            }
            printf("not present");
        }

        //main
        int main()
        {
            //declarations
            int n,i,j,key;

            scanf("%d",&n);
            
            //2d array input
            int a[n][n];
            printf("enter array elements:");

                for(j=0; j<n; j++)
                {
                    scanf("%d",&a[i][j]);
                }
            }

            //key input
