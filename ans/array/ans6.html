//Print all non repeating characters in a string inputted from user

            #include <stdio.h>
                int main(){
                    char s[50];
                    puts("enter string: ");
                    gets(s);
                    int i,j,c;
                    for(i=0;s[i]!='\0';i++)
                    {
                        c=0;
                        for(j=0;s[j]!='\0';j++)
                        {
                            if(s[i]==s[j])
                                c++;
                        }
                        if(c==1)
                            printf("%c ",s[i]);
                    }
                    return 0;
                }
