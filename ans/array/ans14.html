/*Given a matrix of size n X n containing positive integers, write a
            program to rotate the elements of matrix in clockwise direction.*/

            #include <stdio.h>
                void rotate(int n,int a[n][n],int l)
                {
                    int temp=a[l][l],i,j;
                    n=n-l;
                
                    //left side 
                    for(i=l;i<n-1;i++)
                    {
                        a[i][l]=a[i+1][l];
                    }
                
                    //bottom
                    for(j=l;j<n-1;j++)
                    {
                        a[n-1][j]=a[n-1][j+1];	
                    }
                
                    //right side
                    for(i=n-1;i>l;i--)
                    {
                        a[i][n-1]=a[i-1][n-1];
                    }
                
                    //top
                    for(j=n-1;j>l;j--)
                    {
                        a[l][j]=a[l][j-1];
                    }
                    a[l][l+1]=temp;
                }
                
                int main()
                {	
                    int n,i,j,temp,layer;
                    scanf("%d",&n);
                    layer=n/2;
                    
                    //array input
                    int a[n][n];
                    for(i=0;i<n;i++)
                    {
                        for(j=0;j<n;j++)
                        {
                            scanf("%d",&a[i][j]);	
                        }
                    }
                    
                    for(i=0;i<layer;i++)
                        rotate(n,a,i);
                    
                    //printing result
                    for(i=0;i<n;i++)
                    {
                        for(j=0;j<n;j++)
                        {
                            printf("%d ",a[i][j]);	
                        }
                        printf("\n");
                    }
                
                    return 0;
                }
