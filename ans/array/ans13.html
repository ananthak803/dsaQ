/*Given a boolean matrix (contains only 0 and 1) of size m X n where each row is
            sorted, write a a program to find which row has maximum number
            of 1's. */

            #include <stdio.h>
            void main()
            {
                //declarations
                int m,n,i,j,count,max=-1,row=-1;

                //array size input
                printf("enter number of rows and columns:");
                scanf("%d%d",&m,&n);

                //array input
                int a[m][n];
                printf("enter array elements: ");
                for(i=0;i<m;i++)
                {
                    for(j=0;j<n;j++)
                    {
                        scanf("%d",&a[i][j]);
                    }
                }

                // finding row with maximum 1
                for(i=0;i<m;i++)
                {
                    j=n-1;
                    count=0;
                    while(a[i][j]==1 && j>=0)
                    {
                        count++;
                        j--;
                    }
                    if(max<count && count!=0)
                    {
                        max=count;
                        row=i;
                    }
                }

                //printing result
                if(row!=-1)
                    printf("row-%d",row+1);
                else
                    printf("not present");
            
            }
