//Find  all elements in an array that occur an odd number of times.
        
            #include <stdio.h>
            int main()
            {
                //user input
                int n,max;
                printf("enter n:");
                scanf("%d",&n);
                int a[n];
                printf("enter elements of array:");
                for(int i=0; i<n; i++)
                {
                    scanf("%d",&a[i]);
                }
                //finding largest number
                max=a[0];
                for(int i=0; i<n; i++)
                {
                    if(max<a[i])
                    {
                        max=a[i];
                    }
                }
                //declaring freq array and initializing to 0
                int freq[max+1];   //freq[max] not large enough to store the frequency of the maximum element coz index start from 0 . max+1 insted of max
                for(int i=0; i<=max; i++)
                {
                    freq[i]=0;
                }
                //increments in frequency array for all indexes where index are equal to numbers of array a 
                for(int i=0; i<n; i++)
                {
                   freq[a[i]]++;
                }
                //checking for odd occurrences and printing them
                for(int i=0;i<=max; i++)
                {
                    if(freq[i]%2!=0)
                    {
                        printf("%d occuring %d times.",i,freq[i]);
                    }
                }
                return 0;
            }
