/*sort an integer array with n objects colored red, white and blue.
            Sort them in place so that same colors are adjacent, with the colors in order red white blue.
            Use integers 0,1 and 2 to represent the colors red,white and blue respectively.*/
            
            /*Leetcode link : https://leetcode.com/problems/sort-colors/description/  */
            
            //straight forward approach
            #include <stdio.h>
            int main()
            {   //declarations
                int n,i,temp,j,k;
            
                //input array size
                printf("enter size of array:");
                scanf("%d",&n);
                int array[n];
            
                //array input
                printf("enter array elements(0,1,2):");
                for(i=0;i<n;i++)
                {
                    scanf("%d",&array[i]);
                }
            
                //sorting colors
                k=0;                 
                for(i=0;i<3;i++)    //i<3 coz need to sort only three colors 0,1 and 2
                {
                    for(j=0;j<n;j++)
                    {
                        if(array[j]==i)
                        {
                            temp=array[k];
                            array[k++]=array[j];
                            array[j]=temp;
                        }
                    }
                }
            
                //printing result
                for(i=0;i<n;i++)
                {
                    printf("%d ",array[i]);
                }
            }
