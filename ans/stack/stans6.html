/* program to find result of postfix expression. */
        /*note - input should not have spaces and should contain only single digit numbers.*/

            #include <stdio.h>
            #define max 20
                
                void push(int a[], int *top, int c)
                {
                    if (*top == max - 1)
                        return;
                    else
                        a[++(*top)] = c;
                }
                
                int pop(int a[], int *top)
                {
                    if (*top == -1)
                    {
                        printf("empty\n");
                        return 0;
                    }
                    else
                        return a[(*top)--];
                } 
                
                int main()
                {
                    char s[max];
                    int a[max];
                    int top = -1, i = 0,x,y;
                    printf("enter expression:");
                    fgets(s, max, stdin);
                    while (s[i] != '\0')
                    {
                        if (s[i] >='0' && s[i] <='9')
                            push(a, &top, s[i] - '0');
                        else
                        {
                            if(top<1)
                                break;
                            }
                           else
                           {
                                y = pop(a, &top);
                                x = pop(a, &top);
                           }
                
                            switch (s[i])
                            {
                                case '+':
                                    push(a, &top, x + y);
                                    break;
                                case '-':
                                    push(a, &top, x - y);
                                    break;
                                case '/':
                                    push(a, &top, x / y);
                                    break;
                                case '*':
                                push(a, &top, x * y);
                                break;
                            }
                        }
                        i++;
                    if(top!=-1)
                        printf("%d ", pop(a, &top));
                    else
                        printf("no result");
                }
