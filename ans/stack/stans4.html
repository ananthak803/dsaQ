//reversing string using stack(without recursion)

            #include <stdio.h>
            #define max 20
                
                void push(char a[], int *top, char item)
                {
                    if (*top == max - 1)
                        printf("stack full");
                    else
                        a[++*top] = item;
                }
                
                char pop(int *top)
                {
                    if (*top == -1)
                        printf("stack empty");
                    else
                        *top--;
                }
                
                int main()
                {
                    char s[max], a[max];
                    int i, top = -1;
                    printf("enter stack");
                    scanf("%[^\n]s", s);
                    for (i = 0; s[i] != '\0'; i++)
                        push(a, &top, s[i]);
                    i = 0;
                    while (top != -1)
                        s[i++] = a[top--];
                    printf("%s", s);
                }
