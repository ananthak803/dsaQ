/*Find and print majority element in an array.
            The majority element is the element that appears more than ⌊n / 2⌋ times.*/
            
            /*Leetcode link : https://leetcode.com/problems/majority-element/description/  */
            
            //straight forward approach
            #include <stdio.h>
            void main()
            {
                //declaration
                int i,j,count,n;
            
                //size of array input
                printf("enter size of array:");
                scanf("%d",&n);
            
                //arrya input
                int a[n];
                printf("enter array elements:");
                for(i=0;i<n;i++)
                    scanf("%d",&a[i]);
            
                //finding majority number
                for(i=0;i<n;i++)
                {
                    count=0;
                    for( j=0;j<n;j++)
                    {
                        if(a[i]==a[j])
                        {
                            count++;
                        }
                    }
                    if(count>n/2)
                    {
                        //printing result and exiting loop
                        printf("majority element is %d",a[i]);
                        break;            //assuming there is only one majority element
                    }
                }
                printf("no majority element found");
            }
