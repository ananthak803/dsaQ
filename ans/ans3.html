//Second largest element in an array without sorting

            #include <stdio.h>
            #include <limits.h>
            int main(){
                //inputing array
                int n;
                printf("enter n:");   //n should be greater than 2
                scanf("%d",&n);
                int a[n],max=INT_MIN,max2=INT_MIN;  //INT_MIN- min value that can be held by an int type
                printf("enter array elements: ");
                for(int i=0; i<n; i++)
                {
                    scanf("%d",&a[i]);
                }

                //finding 2nd largest number
                for(int i=0; i<n; i++)
                {
                    if(max<a[i])
                    {
                        max=a[i];
                    }
                    if(a[i]<max && a[i]>max2)
                    {
                        max2=a[i];
                    }
                }

                //printing result
                printf("2nd largest element is:%d",max2);
                return 0;
                }
